Uygulamanın bilgileri değişkenlere atıldı. 

[source,]
----
function loadLogin() {
  return fs.readFileSync('view/login.html').toString();
}

app.get('/', function(request, response){
  var view = {
    appId: app_id,
    csrf: csrf_guid,
    version: api_version,
  };

  var html = Mustache.to_html(loadLogin(), view);
  response.send(html);
});
----
